# 1. Hello World

// producer.js

const amqplib = require('amqplib')

async function produce(q = 'tasks', msg = 'something to do') {
    try {
        const conn = await amqplib.connect('amqp://localhost')
        const ch = await conn.createChannel()
        await ch.assertQueue(q)
        ch.sendToQueue(q, Buffer.from(msg))
        console.log('send ===>',q,'==>',msg)
    } catch (error) {
        console.log(error)
    }
}

// consumer.js

const amqplib = require('amqplib')

        const ch = await conn.createChannel()
        await ch.assertQueue(q)

                console.log(msg.content.toString());

        console.log(error)

# 3. Publish/Subscribe

// publisher.js

const getConn = require('./get-connection')

async function publish(exchange = 'pubsub_test', msg = 'logs') {
    try {
        const conn = await getConn()
        const ch = await conn.createChannel()
        await ch.assertExchange(exchange, 'fanout', '')

        const q1 = 'pubsub_queue1'
        const q2 = 'pubsub_queue2'
        await ch.assertQueue(q1)
        await ch.assertQueue(q2)

        await ch.bindQueue(q1, exchange)
        await ch.bindQueue(q2, exchange)

        for(let i =0; i<10; i++){
            ch.publish(exchange, '', Buffer.from(i+msg))
            console.log('send ===>',exchange,'==>',i+msg)
        }

    } catch (error) {
        console.log(error)
    }
}

# 4. Routing

// publisher.js
const getConn = require('./get-connection')

async function publish(exchange = 'routing_test', msg = 'logs') {
    try {
        const conn = await getConn()
        const ch = await conn.createChannel()
        await ch.assertExchange(exchange, 'direct')

        const q1 = 'routing_queue1_error'
        const q2 = 'routing_queue2_other'
        await ch.assertQueue(q1)
        await ch.assertQueue(q2)

        await ch.bindQueue(q1, exchange, 'error')
        await ch.bindQueue(q2, exchange, 'info')
        await ch.bindQueue(q2, exchange, 'warning')
        await ch.bindQueue(q2, exchange, 'error')

        ch.publish(exchange, 'error', Buffer.from('err log body...'))
        console.log('send ===>',exchange,'routingkey: error','==>','err log body...')

        ch.publish(exchange, 'info', Buffer.from('err log body...'))
        console.log('send ===>',exchange,'routingkey: info','==>','err log body...')
        ch.publish(exchange, 'info', Buffer.from('info log body...'))
        console.log('send ===>',exchange,'routingkey: info','==>','info log body...')

    } catch (error) {
        console.log(error)
    }
}

# 5. Topics

  • *通配符可以匹配单级字符
  • #通配符可以匹配多级字符
// publisher.js

const getConn = require('./get-connection')

async function publish(exchange = 'topic_test', msg = 'logs') {
    try {
        const conn = await getConn()
        const ch = await conn.createChannel()
        await ch.assertExchange(exchange, 'topic')

        const q1 = 'topic_queue1_important'
        const q2 = 'topic_queue2_other'
        await ch.assertQueue(q1)
        await ch.assertQueue(q2)

        await ch.bindQueue(q1, exchange, '*.error')
        await ch.bindQueue(q1, exchange, 'order.#')
        await ch.bindQueue(q2, exchange, '#.#')

        ch.publish(exchange, 'user.error', Buffer.from('user.error log body...'))
        console.log('send ===>',exchange,'user.error log body...')

        ch.publish(exchange, 'order.info', Buffer.from('order.info log body...'))
        console.log('send ===>',exchange,'==>','order.info log body...')

        ch.publish(exchange, 'user.info', Buffer.from('user.info log body...'))
        console.log('send ===>',exchange,'user.info log body...')

    } catch (error) {
        console.log(error)
    }
}
